function tryupdateorrelief(p1, p2) {
    //update parent id and index
    if (p2.d > p1.d + distance(p1, p2)) {
        p2.d = p1.d + distance(p1, p2);
        p2.parent = p1.id;
        p2.parentindex = p1.index;
    }
}
